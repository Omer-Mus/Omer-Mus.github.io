145 std::vector<std::vector<std::vector<double>>> simulate(std::vector<Spring>& springs, std::vector<Mass> &masses)
146 {
147
148 int c=0;
149 //FRICTION HERE
150// createFriction();
151
152 //Create a vector that will hold the new positiosn of all the masses.
153 for (auto& m : masses) {
154 std::vector<std::vector<double>> posVec;
155 std::vector<double> initPos = { m.p[0], m.p[1], m.p[2] };
156 posVec.push_back(initPos);
157 massPositions.push_back(posVec);
158 }
159
160 int i = 0;
161 for (auto& m : masses)
162 balls.push_back(KOL_sim::Ball(++i, m.radius, m.color, m.position()[0], m.position()[2], m.position()[1]));
163
164
165
167 while (T < lim) {
168 /**************************************************
169 Part 4.a.
170 **************************************************/
171 //compute the F forces
172 for (auto& m : masses) {
173
174 if(this->externalFroces) {
175 m.add_external_force(externalFrocesSim);
176 }
177 std::vector<double> temp1;
178
179 for (int i=0; i<3; ++i)
180 temp1.push_back(m.m * GRAVITY[i]);
181 for (int i=0; i<3; ++i)
182 m.F[i] += temp1[i];
183
184
185
186 /* TODO: FRICITON
187
188 if (m.p.back() == 0) {
189 Apply Friction....
190 }
191
192 */
193 //apply ground force
194 if (m.p.back() < 0) {
195 std::vector<double> c_pos = {0,0,100'000*m.p[2]*(-1)};
196 for (int i=0; i<3; i++)
197 m.F[i] += c_pos[i];
198 }
199 }
200
201 //iterate over the springs
202 for (auto& s : springs) {
203
204 if (s.actuate_spr) //actuate springs
205 s.actuate_spring(T);
206
207 double f = s.spring_force();
208 std::vector<double> temp;
209 double magnitude = 0;
210 for (int i=0; i<3; i++) {
211 temp.push_back(s.m2->p[i]-s.m1->p[i]);
212 magnitude += pow(s.m1->p[i]-s.m2->p[i], 2);
213 }
214 std::vector<double> tempUnit(temp);
215 for (int i=0; i<3; i++) {
216 if (magnitude <= 0) std::cout << magnitude << "\n";
217 else tempUnit[i] /= sqrt(magnitude); // move outside
218 }
219 for (int i=0; i<3; i++) {
220 tempUnit[i] *= f;
221 s.m1->F[i] -= tempUnit[i];
222 s.m2->F[i] += tempUnit[i];
223 }
224 }
225 /**************************************************
226 Part 4.b.
227 **************************************************/
228 int mass_append = 1;
229 int massID = 0;
230
231 for (auto& m : masses) {
232
233 m.update_acceleration();
234 m.update_velocity(DT);
235 m.update_position(DT);
236
237 if (c == 50) {
238 std::vector<double> newPos = { m.p[0]+3, m.p[1]+3, m.p[2]};
239 massPositions[massID].push_back(newPos);
240 massID+=1;
241 }
242
243 mass_append++;
244 m.update_damping();
245 }
246 T += DT;
247
248 if (c==50) c=0;
249 c++;
250
251 for (auto& m : masses)
252 m.F = (std::vector<double>) {0, 0, 0};
253
254 for (auto& m : masses) {
255 std::vector<std::vector<double>> posVec;
256 std::vector<double> initPos = { m.p[0], m.p[1], m.p[2] };
257 posVec.push_back(initPos);
258 massPositions.push_back(posVec);
259 }
260 for (auto& m : masses)
261 balls.push_back(Ball(++i, m.radius, m.color, m.position()[0], m.position()[2], m.position()[1]));
262
263 } // end Simulation Loop
264
265 return massPositions;
266
267 }
